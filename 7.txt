import java.util.ArrayList;
import java.util.Arrays;
import java.util.Scanner;

public class ExamScheduling {

    static void greedyColoring(ArrayList<ArrayList<Integer>> graph, int numCourses) {
        int[] result = new int[numCourses];
        Arrays.fill(result, -1);
        
        // Assign the first color to the first vertex
        result[0] = 0;

        // A temporary array to store the available colors. 
        // True value of available[cr] would mean that the color cr is assigned to one of its adjacent vertices
        boolean[] available = new boolean[numCourses];
        Arrays.fill(available, true);

        for (int u = 1; u < numCourses; u++) {
            // Process all adjacent vertices and mark their colors as unavailable
            for (int adj : graph.get(u)) {
                if (result[adj] != -1) {
                    available[result[adj]] = false;
                }
            }

            // Find the first available color
            int color;
            for (color = 0; color < numCourses; color++) {
                if (available[color]) {
                    break;
                }
            }

            result[u] = color; // Assign the found color

            // Reset the values back to true for the next iteration
            Arrays.fill(available, true);
        }

        // Output Results
        System.out.println("Exam Slot Assignment (Greedy Coloring):");
        int maxColor = 0;
        for (int u = 0; u < numCourses; u++) {
            System.out.println("Course " + u + " â†’ Slot " + result[u]);
            if (result[u] > maxColor) {
                maxColor = result[u];
            }
        }

        System.out.println("\nTotal Exam Slots Used: " + (maxColor + 1));
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter number of courses: ");
        int numCourses = sc.nextInt();

        System.out.print("Enter number of edges (conflicts): ");
        int edges = sc.nextInt();

        // Initialize Graph
        ArrayList<ArrayList<Integer>> graph = new ArrayList<>(numCourses);
        for (int i = 0; i < numCourses; i++) {
            graph.add(new ArrayList<>());
        }

        System.out.println("Enter edges (u v):");
        for (int i = 0; i < edges; i++) {
            int u = sc.nextInt();
            int v = sc.nextInt();
            
            // Undirected Graph
            graph.get(u).add(v);
            graph.get(v).add(u);
        }

        greedyColoring(graph, numCourses);
        
        sc.close();
    }
}
